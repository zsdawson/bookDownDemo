#dplyr to combine pseudoreplicates and then obtain n, mean and sem

# Example data
set.seed(123)
data <- data.frame(
  replicate = rep(1:3, each = 3),
  value = rnorm(9, mean = 10, sd = 2),
  pseudo = rep(1:3, 3)
)
# Combine pseudoreplicates
combined_data <- data %>%
  group_by(replicate) %>%
  summarise(n = n(),
            mean = mean(value),
            sd = sd(value),
            sem = sd/sqrt(n)) %>%
  ungroup()

# View combined data
